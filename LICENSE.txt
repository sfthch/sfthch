VehicleDetail_getVehicleAlarmStatus

SELECT
			a.VEH_ID                                        			AS 'vehId',
			a.alarm_id 													AS 'alarmId',
			a.ALARM_STATE_TPCD 									AS 'alarmStateTpcd',
			ifnull(max(b.react_seq), '') 							AS 'reactSeq',
			(SELECT common_detail_cd_nm
			FROM tb_detailcode
			WHERE common_detail_cd = a.ALARM_STATE_TPCD
			AND common_cd_id ='MON009'
			) 																AS 'alarmState',
			(SELECT common_detail_cd_nm
			FROM tb_detailcode
			WHERE common_detail_cd = a.ALARM_MSG_TPCD
			AND common_cd_id ='MON013'
			) 																AS 'alarmMsgType',
			'-' AS 'userId',
			DATE_FORMAT(a.EVT_DATETIME, '%Y-%m-%d %H:%i:%s') 		AS 'evtDatetime',
			
			(select ifnull(b1.REACT_DATETIME, '-') from tb_alarm a1			
			inner JOIN tb_alarm_reaction b1 ON a1.ALARM_ID = b1.ALARM_ID
			, tb_vehicle c1
			where a1.VEH_ID = #{vehId}
			and  a1.VEH_ID = c1.VEH_ID
			and a1.ALARM_STATE_TPCD IN ('0', '1')) as  'completeDt',
			
			c.VEH_PLATE_NUM as vehPlateNum
			FROM tb_alarm a
			LEFT OUTER JOIN tb_alarm_reaction b ON a.ALARM_ID = b.ALARM_ID
			, tb_vehicle c
			WHERE a.VEH_ID = #{vehId}
			AND a.VEH_ID = c.VEH_ID
			AND a.ALARM_STATE_TPCD IN ('0', '1')
			GROUP BY alarmId, alarmStateTpcd
			
		UNION ALL
		SELECT
			a.VEH_ID                                        			AS 'vehId',
			a.alarm_id 													AS 'alarmId',
			a.ALARM_STATE_TPCD 									AS 'alarmStateTpcd',
			ifnull(max(b.react_seq), '') 							AS 'reactSeq',
			(SELECT common_detail_cd_nm
			FROM tb_detailcode
			WHERE common_detail_cd = a.ALARM_STATE_TPCD
			AND common_cd_id ='MON009'
			) 																AS 'alarmState',
			(SELECT common_detail_cd_nm
			FROM tb_detailcode
			WHERE common_detail_cd = a.ALARM_MSG_TPCD
			AND common_cd_id ='MON013'
			) 																AS 'alarmMsgType',
			'-' AS 'userId',
			DATE_FORMAT(a.EVT_DATETIME, '%Y-%m-%d %H:%i:%s') 		AS 'evtDatetime',
			
			(select ifnull(b1.REACT_DATETIME, '-') from tb_alarm a1			
			inner join tb_alarm_reaction b1 ON a1.ALARM_ID = b1.ALARM_ID
			, tb_vehicle c1
			where a1.VEH_ID = 'VH_SJA_0000025'
			and  a1.VEH_ID = c1.VEH_ID
			and a1.ALARM_STATE_TPCD IN ('0', '1')) as  'completeDt',			

			c.VEH_PLATE_NUM as vehPlateNum
			FROM tb_alarm a
			LEFT OUTER JOIN tb_alarm_reaction b ON a.ALARM_ID = b.ALARM_ID
			, tb_vehicle c
			WHERE a.VEH_ID = #{vehId}
			AND a.VEH_ID = c.VEH_ID
			AND a.ALARM_STATE_TPCD = '2'
			AND b.REACT_DATETIME Between DATE_ADD(CURDATE(), INTERVAL 2 HOUR) AND DATE_ADD(DATE_ADD(CURDATE(), INTERVAL 1 DAY), INTERVAL 2 HOUR)
			GROUP BY alarmId, alarmStateTpcd
			ORDER BY alarmId
