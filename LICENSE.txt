EVChargingStationDetail_getEVChargerInfo
SELECT 
		EVCHGR_ID as evchgrId
		, EXTL_EVCHGR_ID as extlEvchgrId
		, EVCHGR_TPCD as evchgrTpcd
		, EVCHGR_CPCT_VAL as evchgrCpctVal
		, EVCH_UNPR_ID as evchUnprId
		, EVCHGR_TRCK_CNT as evchgrTrckCnt
		, DCOM_SUPT_YN as dcomSuptYn
		, DCDEM_SUPT_YN as dcdemSuptYn
		, ACTP_SUPT_YN as actpSuptYn
		, SLCHR_SUPT_YN as slchrSuptYn
		, SUCHR_SUPT_YN as suchrSuptYn
		, DSCHR_SUPT_YN as dschrSuptYn
		, 'N' as pmChSuptYn
	FROM
		TB_EVCHARGER 
	WHERE  
		EVCHST_ID = #{ evchstId }
	ORDER BY EXTL_EVCHGR_ID
