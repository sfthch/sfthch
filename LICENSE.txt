MaaSData_getStopPointList

SELECT 
	  		EXTRN_STOPPOINT_ID AS stopId
	  		,STOPPOINT_NM AS stopNm
	  		,X_COORD AS stopX
	  		,Y_COORD AS stopY
	  		,EXTRN_STOPPOINT_ID AS stopNo
	  		,IFNULL(STOPPOINT_TYPE_CODE, '') AS stopTypeCode
	  		,USE_YN as useYn
		FROM TB_STOPPOINT 
		WHERE VER_ID= #{verId}
		AND OPERR_ID = #{operrId}
		AND EXTRN_STOPPOINT_ID IN
				<foreach collection="stoPointIdList" item="arr" open="(" close=")" separator=",">
				 #{arr}			
				</foreach>
		ORDER BY CAST( SUBSTR(STOPPOINT_ID from 8) as signed)
