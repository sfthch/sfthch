TransitNetwork_selectJrnyPattrnIdByExtrnLineId
SELECT C.JRNYPATTRN_ID
		FROM tb_route A
			INNER JOIN TB_LINE B
				ON A.LINE_ID = B.LINE_ID 
			INNER JOIN TB_JRNYPATTRN C 
				ON A.ROUT_ID = C.ROUT_ID 
		WHERE 
			B.OPERR_ID = #{operrId}
		AND
			B.EXTRN_LINE_ID = #{extrnLineId}
		AND
			B.LAST_YN = 'Y'
		AND 
			C.VER_ID = #{verId}
