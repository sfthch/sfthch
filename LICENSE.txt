MaaSData_getVehicleList
SELECT
			EXTRN_VEH_ID AS vehicleId
			,VEH_PLATE_NUM AS vehicleNo
			,IFNULL(VEH_DESC, '') AS vehicleNm
			,LPAD(VEH_STATE_CD, '2', '0') AS vehicleStatusCode
			,USE_YN as useYn
		FROM
			TB_VEHICLE 
		WHERE
			OPERR_ID = #{operrId}
		AND 
			LAST_YN = 'Y'
		AND EXTRN_VEH_ID IN
				<foreach collection="vehIdList" item="arr" open="(" close=")" separator=",">
				 #{arr}			
				</foreach>			
		ORDER BY EXTRN_VEH_ID
