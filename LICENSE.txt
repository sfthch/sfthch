VehJrnyTransaction_getVehJrnyRunStatus
SELECT 
    			VEH_CURR_SPD as vehCurrSpd,
    			VEH_HEADING_ANGLE as vehHeadingAngle,
    			ACCUM_PSNGR_CNT as accumPsngrCnt,
    			CHG_DATETIME as chgDatetime,
    			JRNYPATTRN_ID as jrnypattrnId,
    			VEH_ID as vehId, 
    			TRV_STATE_CD as trvStateCd,
    			X_COORD as xCoord, 
    			Y_COORD as yCoord, 
    			TRIP_SEQ as tripSeq,
    			TRV_START_DATETIME as trvStartDatetime,
    			TRV_END_DATETIME as trvEndDatetime,
    			ACCUM_TRV_DIST as accumTrvDist,
    			ACCUM_TRV_TM as accumTrvTm,
    			TRV_TYPE_CD as trvTypeCd,
    			BATT_REM_AMT as battRemAmt,
    			DOOR_OPEN_YN as doorOpenYn,
    			DRIVE_SERIES_NUM as driveSeriesNum,
    			VER_ID as verId
    	FROM 
    			tb_vehjrnyrunstatus 
    	WHERE 
    			VEH_ID = #{vehId}
