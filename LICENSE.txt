1. selectVehicleByExtrnVehId
		SELECT 
			VEH_ID AS vehId
			,VEH_PLATE_NUM AS vehPlateNum
		FROM TB_VEHICLE 
		WHERE 
			EXTRN_VEH_ID ='9001'		
		AND 
			OPERR_ID = 'OP_SJA_0000003'			

2. selectLineIdByExtrnLineId
		SELECT 
			LINE_ID 
		FROM TB_LINE
		WHERE
			EXTRN_LINE_ID = '1'
		AND
			OPERR_ID ='OP_SJA_0000003'
		AND
			LAST_YN = 'Y'

3. selectMaasSchedule - #1
    SELECT
        LINE_ID
        ,OPER_DAY
        ,PLANNING_ID
        ,VEH_ID
        ,DRIVER_ID
        ,TRIP_START_DATETIME
        ,TRIP_END_DATETIME
        ,JOURNEY_INTVAL
        ,JOURNEY_PATTRN_CD
        ,JOURNEY_PATTRN_CD_NM
        ,USE_YN
        ,DB_REG_DT
        ,USER_ID
    FROM
            tb_vehiclejourneyplanmaas
    WHERE
        LINE_ID = 'LE_SJA_0000087'
    AND
        OPER_DAY = '20220707'
    AND
        PLANNING_ID = '9061'

5. retrieveSiteId
SELECT	
    A.OPERR_ID as operrId
    ,A.SITE_ID as siteId
    ,A.OPERR_SHORT_NM as operrShortNm
    ,A.USER_ID as userId
    ,A.SYS_CLASS_CD as systemCode
    ,A.SVC_CLASS_CD as serviceCode
    ,A.AFFICMPNY_CD as affiliateCode
FROM	TB_OPERATOR A, TB_USER B
WHERE A.USER_ID = B.USER_ID
AND	A.SYS_CLASS_CD = '02'
AND	A.SVC_CLASS_CD = '20'
AND	A.AFFICMPNY_CD = '04'

6. SELECT C.JRNYPATTRN_ID
    FROM tb_route A
        INNER JOIN TB_LINE B
            ON A.LINE_ID = B.LINE_ID 
        INNER JOIN TB_JRNYPATTRN C 
            ON A.ROUT_ID = C.ROUT_ID 
    WHERE 
        B.OPERR_ID = 'OP_SJA_0000003'
    AND
        B.EXTRN_LINE_ID = '1'
    AND
        B.LAST_YN = 'Y'
    AND 
        C.VER_ID = 'NV_SJA_0000002'
