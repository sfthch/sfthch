VehicleDetail_getVehicleAlarmStatistics
SELECT
			VEH_ID								AS "vehId",
			ALARM_STATE_TPCD 			AS "alarmStateTpcd",
			COMMON_DETAIL_CD_NM 	AS "alarmType",
			CNT 									AS "count"
		FROM (
			SELECT A.VEH_ID, A.ALARM_STATE_TPCD, COUNT(*) AS CNT
			FROM TB_ALARM A
			WHERE A.VEH_ID = #{vehId}
				AND A.ALARM_STATE_TPCD IN ('0', '1')
			GROUP BY A.ALARM_STATE_TPCD
		UNION ALL
		SELECT A.VEH_ID, A.ALARM_STATE_TPCD, COUNT(*) AS CNT
		FROM TB_ALARM A
		JOIN (
				SELECT
					MAX(REACT_SEQ),
					ALARM_ID,
					MAX(REACT_DATETIME) AS 'REACT_DATETIME'
				FROM TB_ALARM_REACTION
				GROUP BY ALARM_ID
			) B ON A.ALARM_ID = B.ALARM_ID
			WHERE B.REACT_DATETIME BETWEEN DATE_ADD(CURDATE(), INTERVAL 2 HOUR) AND DATE_ADD(DATE_ADD(CURDATE(), INTERVAL 1 DAY), INTERVAL 2 HOUR)
			  	AND A.VEH_ID = #{vehId}
			  	AND A.ALARM_STATE_TPCD = '2'
			GROUP BY A.ALARM_STATE_TPCD
		) A
		JOIN (
			SELECT
				COMMON_DETAIL_CD,
				COMMON_DETAIL_CD_NM
			FROM TB_DETAILCODE
			WHERE COMMON_CD_ID ='MON009'
		) B ON A.ALARM_STATE_TPCD=B.COMMON_DETAIL_CD
