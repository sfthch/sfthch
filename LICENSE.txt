VehicleDetail_getCurrentStatus
SELECT
			a.VEH_ID														AS 'vehId',
			a.VEH_PLATE_NUM                                           		AS 'vehPlateNum',
			b.MODEL_NUM 													AS 'vehNo',
			IFNULL(g.LINE_ID, '')															AS 'lineId',
			IFNULL(g.LINE_NM, '')															AS 'lineNm',
			CONCAT(b.VEH_TYPE_NM, ' | ', a.VEH_ID) 	AS 'vehDesc',
			IFNULL(a.VEH_STATE_CD, '1')								AS 'opStateCd',
			(
				SELECT common_detail_cd_nm
				FROM tb_detailcode
				WHERE common_detail_cd = IFNULL(a.VEH_STATE_CD, '1')
				AND common_cd_id ='VEH002'
			) 																		AS 'opStateNm',
			IFNULL(d.TRV_STATE_CD, '0') 												AS 'vehStateCd',
			IFNULL((
				SELECT common_detail_cd_nm
				FROM tb_detailcode
				WHERE common_detail_cd = d.TRV_STATE_CD
				AND common_cd_id ='MON001'
			), 'OFF')																		AS 'vehStateNm',
			b.VEH_TYPE_NM 													AS 'vehType',
			IFNULL(a.MNFCTUR_DATE, '') 								AS 'mnfcturYear',
			IFNULL(a.VEH_PLATE_NUM, '') 								AS 'plateNo',
			IFNULL(b.MODEL_NUM, '') 									AS 'modelNo',
			c.GARAGE_ID														AS 'depotId',
			c.GARAGE_NM 													AS 'depotNm',
			IFNULL(d.TRV_TYPE_CD,'')													AS 'trvTypeCd',
			IFNULL((
				SELECT common_detail_cd_nm
				FROM tb_detailcode
				WHERE common_detail_cd = d.TRV_TYPE_CD
				AND common_cd_id = 'MON015'
			),'Undefined') 																		AS 'trvTypeNm',
			h.STRM_URL                                                       AS 'strmUrl'
		FROM tb_vehicle a
		LEFT JOIN tb_typeofveh b ON a.VEH_TYPE_ID = b.VEH_TYPE_ID
		LEFT JOIN tb_garage c ON a.GARAGE_ID = c.GARAGE_ID
		LEFT JOIN tb_vehjrnyrunstatus d ON a.VEH_ID = d.VEH_ID
		LEFT JOIN tb_jrnypattrn e ON d.JRNYPATTRN_ID = e.JRNYPATTRN_ID AND d.VER_ID = e.VER_ID
		LEFT JOIN tb_route f ON e.ROUT_ID = f.ROUT_ID AND e.VER_ID = f.VER_ID
		LEFT JOIN tb_line g ON g.LINE_ID = f.LINE_ID AND g.VER_ID = f.VER_ID
		LEFT OUTER JOIN tb_camerastatus h ON a.VEH_ID = h.VEH_ID
		WHERE a.VEH_ID = #{vehId}
