MonitoringMap_getVehicleDetailForPopup

SELECT A.VEH_ID as vehId, B.EXTRN_VEH_ID as extrnVehId,  B.VEH_PLATE_NUM as vehPlateNum,
			   E.LINE_NM as lineNm, E.LINE_ID as lineId, E.EXTRN_LINE_ID  as extrnLineId,
			   F.GARAGE_ID as garageId, F.EXTRN_GARAGE_ID as extrnGarageId, F.GARAGE_NM as garageNm, 
			   A.ACCUM_PSNGR_CNT as accumPsngrCnt, G.PSNGR_CAPA as psngrCapa,
		       A.BATT_REM_AMT/G.BATT_CAPA*100 as battRemPcnt, A.VEH_CURR_SPD as vehCurrSpd, A.TRV_STATE_CD as trvStateCd
		       		       
		  FROM TB_VEHJRNYRUNSTATUS A, TB_VEHICLE B,
		       TB_JRNYPATTRN C, TB_ROUTE D, TB_LINE E, TB_GARAGE F,
			   TB_TYPEOFVEH G
		WHERE A.VEH_ID = B.VEH_ID
		  AND A.JRNYPATTRN_ID = C.JRNYPATTRN_ID AND A.VER_ID = C.VER_ID
		  AND C.ROUT_ID = D.ROUT_ID AND C.VER_ID = D.VER_ID
		  AND D.LINE_ID = E.LINE_ID AND D.VER_ID = E.VER_ID
		  AND B.GARAGE_ID = F.GARAGE_ID
		  AND B.VEH_TYPE_ID = G.VEH_TYPE_ID
		  AND A.VEH_ID=#{vehId}
