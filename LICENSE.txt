VehicleDetail_getVehicleDetailStatus
SELECT
			a.VEH_ID 																				AS 'vehId',
			ifnull(d.ACCUM_PSNGR_CNT, 0) 												AS PSNGR_CNT,
			c.PSNGR_CAPA																		AS PSNGR_CAPA,
			ROUND((4 * ifnull(d.ACCUM_PSNGR_CNT, 0) / c.PSNGR_CAPA), 1) 	AS PSNGR_RATIO,
			ifnull(d.BATT_REM_AMT, 0) 														AS BATT_LV,
			ifnull(c.BATT_CAPA, 100) 															AS BATT_CAPA,
			ifnull(d.DOOR_OPEN_YN, '1') 												AS DOOR_OPEN_YN,
			ifnull((
				SELECT common_detail_cd_nm
				FROM tb_detailcode
				WHERE  common_detail_cd = d.DOOR_OPEN_YN AND common_cd_id ='MON016'
			), 'Door Close') 															AS DOOR_OPEN_STATUS,
			ifnull(d.VEH_CURR_SPD, 0)  																	AS SPEED,
			ifnull(ADAS_REAR_LDR, '00') 													AS ADAS_REAR_LDR,
			ifnull(ADAS_INTRR_CAM_1, '00') 												AS ADAS_INTRR_CAM_1,
			ifnull(ADAS_INTRR_CAM_2, '00') 												AS ADAS_INTRR_CAM_2,
			ifnull(ADAS_INTRR_TEMPRT_SNSR, '00') 										AS ADAS_INTRR_TEMPRT_SNSR,
			ifnull(ADAS_GNSS_STATUS, '00') 												AS ADAS_GNSS_STATUS,
			ifnull(ADAS_WHEEL_SPD_SNSR_FL, '00') 									AS ADAS_WHEEL_SPD_SNSR_FL,
			ifnull(ADAS_WHEEL_SPD_SNSR_FR, '00') 									AS ADAS_WHEEL_SPD_SNSR_FR,
			ifnull(ADAS_WHEEL_SPD_SNSR_RL, '00') 									AS ADAS_WHEEL_SPD_SNSR_RL,
			ifnull(ADAS_WHEEL_SPD_SNSR_RR, '00') 									AS ADAS_WHEEL_SPD_SNSR_RR,
			ifnull(ADAS_SUSP_ROTATN_SNSR_F, '00') 									AS ADAS_SUSP_ROTATN_SNSR_F,
			ifnull(ADAS_SUSP_ROTATN_SNSR_R, '00') 									AS ADAS_SUSP_ROTATN_SNSR_R,
			ifnull(ADAS_FWD_LDR_L, '00') 													AS ADAS_FWD_LDR_L,
			ifnull(ADAS_FWD_LDR_M, '00') 													AS ADAS_FWD_LDR_M,
			ifnull(ADAS_FWD_LDR_R, '00') 													AS ADAS_FWD_LDR_R,
			ifnull(ADAS_TRFSIGN_RECOG_CAM, '00') 									AS ADAS_TRFSIGN_RECOG_CAM,
			ifnull(ADAS_EXTRR_TEMPRT_SNSR, '00') 										AS ADAS_EXTRR_TEMPRT_SNSR,
			ifnull(ADAS_WIN_FOG_SNSR_1, '00') 											AS ADAS_WIN_FOG_SNSR_1,
			ifnull(ADAS_WIN_FOG_SNSR_2, '00') 											AS ADAS_WIN_FOG_SNSR_2,
			ifnull(ADAS_ODM_SNSR_FL, '00') 												AS ADAS_ODM_SNSR_FL,
			ifnull(ADAS_ODM_SNSR_FR, '00') 												AS ADAS_ODM_SNSR_FR,
			ifnull(ADAS_ODM_SNSR_RL, '00') 												AS ADAS_ODM_SNSR_RL,
			ifnull(ADAS_ODM_SNSR_RR, '00') 												AS ADAS_ODM_SNSR_RR,
			ifnull(ADAS_IMPACT_SNSR_F, '00') 											AS ADAS_IMPACT_SNSR_F,
			ifnull(ADAS_IMPACT_SNSR_R, '00') 											AS ADAS_IMPACT_SNSR_R,
			ifnull(ADAS_SUSP_HEIGHT_SNSR_LFT, '00') 								AS ADAS_SUSP_HEIGHT_SNSR_L,
			ifnull(ADAS_SUSP_HEIGHT_SNSR_RGHT, '00') 								AS ADAS_SUSP_HEIGHT_SNSR_R
		FROM tb_vehicle a
		LEFT JOIN tb_connautovehstatus b ON a.VEH_ID = b.VEH_ID
		LEFT JOIN tb_typeofveh c ON a.VEH_TYPE_ID = c.VEH_TYPE_ID
		LEFT JOIN tb_vehjrnyrunstatus d ON a.VEH_ID = d.VEH_ID
		WHERE a.VEH_ID = #{vehId}
