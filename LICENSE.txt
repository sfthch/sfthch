select cast(a.useTime as signed integer) as time, a.cnt as booked, a.cnt as competed, '0' sameDaylastWeek from
		(
			SELECT
			SUBSTR(DATE_FORMAT(A.USE_DT, '%Y%m%d%H'), 9, 2) useTime
			,SUM(A.PSGER_CNT) cnt
			FROM TB_RIDEDATA A, TB_OPERATOR B			
			WHERE A.OPERR_ID = B.OPERR_ID 
			AND B.SYS_CLASS_CD = '02'
			AND DATE_FORMAT(A.USE_DT, '%Y%m%d') =  #{operStartDate}
			GROUP BY 
			SUBSTR(DATE_FORMAT(A.USE_DT, '%Y%m%d%H'), 9, 2)
			ORDER BY 
			SUBSTR(DATE_FORMAT(A.USE_DT, '%Y%m%d%H'), 9, 2)
