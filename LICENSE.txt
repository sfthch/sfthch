TransitNetwork_getTimingLinkList
SELECT TIMING_LINK_ID as timingLinkId, TIMING_LINK_ID as name, START_TIMING_POINT_ID as startTimingPointId, 
			   B.TIMING_POINT_TPCD as startTimingPointTpcd, B.STOPPOINT_ID as startStopPointId, 
			   C.TIMING_POINT_TPCD as endTimingPointTpcd, C.STOPPOINT_ID as endStopPointId,
		       END_TIMING_POINT_ID as endTimingPointId,  TIMING_LINK_LEN as timingLinkLen,
		       A.VER_ID as verId
		  FROM tb_timinglink A, tb_timingpoint B, tb_timingpoint C   
		 WHERE A.START_TIMING_POINT_ID = B.TIMING_POINT_ID
		   AND A.END_TIMING_POINT_ID = C.TIMING_POINT_ID
		   AND A.VER_ID = B.VER_ID 
		   AND A.VER_ID = C.VER_ID
		   AND B.VER_ID=#{verId}
		   AND C.VER_ID=#{verId} 
	  ORDER BY CAST( SUBSTR(TIMING_LINK_ID from 8) as signed)
