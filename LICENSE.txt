DashBoard_getAlarmAbnormalStatus
SELECT ALM.ALARM_STATE_TPCD AS "stateCd"
		     , (SELECT COMMON_DETAIL_CD_NM FROM tb_detailcode WHERE COMMON_CD_ID ='MON009' AND COMMON_DETAIL_CD = ALM.ALARM_STATE_TPCD) AS "alarmType"
		     , IFNULL(COUNT(ALM.ALARM_STATE_TPCD), '0') AS "count"
		  FROM TB_ALARM ALM
		     , TB_DETAILCODE DCD
		 WHERE ALM.SITE_ID = #{siteId}
		   AND ALM.EVT_DATETIME BETWEEN STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i') AND DATE_ADD(STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i'), INTERVAL 1 DAY)
		   AND DCD.COMMON_CD_ID = 'MON013'
		   AND DCD.COMMON_DETAIL_CD_NM LIKE 'ABNORMAL_%'
		   AND ALM.ALARM_MSG_TPCD = DCD.COMMON_DETAIL_CD
		 GROUP BY ALM.ALARM_STATE_TPCD
