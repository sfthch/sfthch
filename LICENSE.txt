VehJrnyTransaction_selectVehicleLocation
SELECT
			'' as reserveId
			,(SELECT EXTRN_LINE_ID FROM TB_LINE WHERE OPERR_ID =  #{operrId} AND LINE_ID =  #{lineId}) as routeId
			,(SELECT EXTRN_VEH_ID FROM TB_VEHICLE WHERE OPERR_ID =  #{operrId} AND VEH_ID = #{vehId}) as vehicleId
			,(SELECT VEH_PLATE_NUM FROM TB_VEHICLE WHERE OPERR_ID =  #{operrId} AND VEH_ID = #{vehId}) as vehicleNo
			
			,REM_SEAT_CNT as remainSeatCnt
			,VEH_CNGST_STATE as vehCngstState

			,(SELECT B.EXTRN_STOPPOINT_ID as departStopId FROM TB_VEHICLELOCATION A INNER JOIN
			TB_STOPPOINT B
			ON A.PRV_STOPPOINT = B.STOPPOINT_ID
			) as departStopId
			
			,PRV_STOPPOINT_IDX as departStopIndex		
			,PRV_STOPPOINT_ARRI_DT as  departStopArriveDtime
			
			,NEXT_STOPPOINT_ID as nextStopId
			,NEXT_STOPPOINT_IDX as nextStopIndex
			,NEXT_STOPPOINT_DIST as nextStopDistnace
			,NEXT_STOPPOINT_ARRIV_DT as nextStopArriveDtime
			
			,XCOORD as vehicleX
			,YCOORD as vehicleY
			
			,REAL_JOURNEY_START_DT as drvStartDtime
			,REAL_JOURNEY_END_DT as drvEndDtime
			,PLAINING_ID as allocateCarId
		FROM
			TB_VEHICLELOCATION 
		WHERE
			OPERR_ID = #{operrId}
		AND
			VEH_ID =  #{vehId}
