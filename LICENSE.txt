	SELECT	
	/* mrserver.mapper.reservation.ReservationUseMapper.inquiryReservationUseDetailAvDrList */
			A.RSEV_UTLZ_SQNO					AS RSEV_UTLZ_SQNO
	,		(
			SELECT Y.CMMN_CD_NM
			FROM TBMSM311 X
			INNER JOIN TBMSM312 Y ON X.CMMN_CD_GRP_SQNO = Y.CMMN_CD_GRP_SQNO
			WHERE 1 = 1
			AND X.USE_YN = 'Y'
			AND Y.USE_YN = 'Y'
			AND X.CMMN_CD_GRP_ID = "RSEV_UTLZ_CLSS_CD"
			AND Y.CMMN_CD_ID = A.RSEV_UTLZ_CLSS_CD
			) AS USE_CLASS
	,		DATE_FORMAT(A.RSEV_UTLZ_DTTM, '%Y-%m-%d %H:%i:%s') AS USE_DTTM
	,		(
			SELECT	STN_NM
			FROM	TBMSM211
	        WHERE	STN_ID = A.RSEV_UTLZ_PLAC_ID
	        AND		MBLT_SYSTM_CLSS_CD = B.MBLT_SYSTM_CLSS_CD
            AND		MBLT_SRVC_CLSS_CD = B.MBLT_SRVC_CLSS_CD
            AND		AFFI_CLSS_CD = B.AFFI_CLSS_CD
	        ) 									AS USE_PLACE
	,		SUM(IFNULL(A.UTLZ_HOUR_VAL,0)) 		AS UTLZ_HOUR_VAL
	,		SUM(ifnull(A.UTLZ_DSTNE_VAL,0)) 	AS UTLZ_DSTNE_VAL
	FROM	TBMSD107 A
	,		TBMSD106 B
	WHERE	1=1
	AND		A.MEMB_SQNO = #{membSqno}
	AND		A.RSEV_SQNO = #{rsevSqno}
	AND		A.RSEV_SQNO = B.RSEV_SQNO
	AND		A.MEMB_SQNO = B.MEMB_SQNO
	AND		A.APP_SCRN_DSPY_YN = 'Y'
	GROUP BY A.RSEV_UTLZ_SQNO
	,		A.RSEV_UTLZ_CLSS_CD
	,		DATE_FORMAT(A.RSEV_UTLZ_DTTM, '%Y-%m-%d %H:%i:%s')
