DashBoard_getVehicleStatusSummary
SELECT 
			"Reg Vehicle Cnt" as "type", 
			IFNULL(COUNT(VEH_ID), 0) as "count"  
		FROM
			TB_VEHICLE a, TB_OPERATOR b
		WHERE A.OPERR_ID = B.OPERR_ID
		AND B.SYS_CLASS_CD = '02'
			
		UNION all 
				
		SELECT 
			"Run Vehicle Cnt" as "type",
			IFNULL(COUNT(A.VEH_ID), 0) as "count"
		FROM
			TB_VEHICLE A, tb_vehjrnyrunstatus B, TB_OPERATOR C
		WHERE
			A.VEH_ID  = B.VEH_ID
		AND
			A.OPERR_ID  = C.OPERR_ID 
		AND
			B.TRV_STATE_CD = 6
		AND
			C.SYS_CLASS_CD  = '02'
			
		UNION all 
		
		SELECT 
			"Abnormal Vehicle Cnt" as "type",
			IFNULL(COUNT(A.VEH_ID), 0) as "count"
		FROM
			TB_VEHICLE A, tb_vehjrnyrunstatus B, TB_OPERATOR C
		WHERE
			A.VEH_ID  = B.VEH_ID
		AND
			A.OPERR_ID  = C.OPERR_ID 			
		AND
			B.TRV_STATE_CD = 9
		AND
			C.SYS_CLASS_CD  = '02'
