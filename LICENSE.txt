VehJrnyTransaction_getJourneyStatus
SELECT 
			JOURNEY_UNQ_NO as journeyUnqNo
			,OPERR_ID as operrId
			,PLANNING_ID as planningId
			,LINE_ID as lineId 
			,PLAN_JOURNEY_VEH_ID as planJourneyVehId
			,PLAN_JOURNEY_DAY as planJourneyDay
			,PLAN_JOURNEY_START_DT as planJourneyStartDt
			,PLAN_JOURNEY_END_DT as planJourneyEndDt
			,REAL_JOURNEY_VEH_ID as realJourneyVehId
			,REAL_JOURNEY_START_DT as realJourneyStartDt
			,REAL_JOURNEY_END_DT as realJoruneyEndDt
			,JOURNEY_STATE_CD as journeyStateCd
			,DB_REG_DT as dbRegDt
		FROM
			TB_JOURNEYSTATUS	
		WHERE	
		  OPER_DAY = #{operDay}
	  AND
	  	VEH_ID = #{vehId}
	  AND
			JOURNEY_UNQ_NO = #{journeyUnqNo}
		AND
			OPERR_ID = #{operrId}
