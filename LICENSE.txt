DashBoard_getEtcStatics
SELECT "alarm veh" as "type"
		     , IFNULL(COUNT(DISTINCT ALM.VEH_ID), 0) AS "count"
		  FROM TB_ALARM ALM
		 WHERE ALM.SITE_ID = #{siteId}
		   AND ALM.ALARM_TPCD = '0'
		   AND ALM.EVT_DATETIME BETWEEN STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i') AND DATE_ADD(STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i'), INTERVAL 1 DAY)
		   AND ALM.ALARM_STATE_TPCD IN ('0', '1')
		UNION ALL
		SELECT "abnormal veh" as "type"
		     , IFNULL(COUNT(DISTINCT ALM.VEH_ID), 0) AS "count"
		  FROM TB_ALARM ALM
		     , TB_DETAILCODE DCD
		 WHERE ALM.SITE_ID = #{siteId}
		   AND ALM.EVT_DATETIME BETWEEN STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i') AND DATE_ADD(STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i'), INTERVAL 1 DAY)
		   AND DCD.COMMON_CD_ID = 'MON013'
		   AND DCD.COMMON_DETAIL_CD_NM LIKE 'ABNORMAL_%'
		   AND ALM.ALARM_STATE_TPCD IN ('0', '1')
		   AND ALM.ALARM_MSG_TPCD = DCD.COMMON_DETAIL_CD
		UNION ALL
		SELECT "malfunction veh" as "type"
		     , IFNULL(COUNT(DISTINCT ALM.VEH_ID), 0) AS "count"
		  FROM TB_ALARM ALM
		     , TB_DETAILCODE DCD
		 WHERE ALM.SITE_ID = #{siteId}
		   AND ALM.EVT_DATETIME BETWEEN STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i') AND DATE_ADD(STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i'), INTERVAL 1 DAY)
		   AND DCD.COMMON_CD_ID = 'MON013'
		   AND DCD.COMMON_DETAIL_CD_NM LIKE 'MALFUNC_%'
		   AND ALM.ALARM_STATE_TPCD IN ('0', '1')
		   AND ALM.ALARM_MSG_TPCD = DCD.COMMON_DETAIL_CD
		UNION ALL
		SELECT "inactive veh" as "type"
		     , IFNULL(COUNT(DISTINCT ALM.VEH_ID), 0) AS "count"
		  FROM TB_ALARM ALM
		     , TB_DETAILCODE DCD
		 WHERE ALM.SITE_ID = #{siteId}
		   AND ALM.EVT_DATETIME BETWEEN STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i') AND DATE_ADD(STR_TO_DATE(#{operStartDateTime},'%Y%m%d%H%i'), INTERVAL 1 DAY)
		   AND DCD.COMMON_CD_ID = 'MON013'
		   AND DCD.COMMON_DETAIL_CD_NM LIKE 'VEH_INACTV_%'
		   AND ALM.ALARM_STATE_TPCD IN ('0', '1')
		   AND ALM.ALARM_MSG_TPCD = DCD.COMMON_DETAIL_CD
