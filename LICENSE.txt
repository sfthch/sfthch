TransitNetwork_getRoutePointForLineList
SELECT distinct A.ROUT_POINT_ID AS name, A.ROUT_POINT_ID AS routPointId, A.X_COORD AS xCoord, A.Y_COORD AS yCoord, D.VER_ID AS verId 
		FROM tb_routepoint A, tb_routelink B, tb_linkonroute C, tb_route D
		WHERE D.VER_ID=#{verId}
	      AND C.LAST_YN = 'Y'
		  AND D.ROUT_ID = C.ROUT_ID AND D.VER_ID = C.VER_ID
		  AND C.ROUT_LINK_ID = B.ROUT_LINK_ID AND C.VER_ID = B.VER_ID
		  AND ((B.START_POINT_ID = A.ROUT_POINT_ID AND B.VER_ID = A.VER_ID) OR (B.END_POINT_ID = A.ROUT_POINT_ID AND B.VER_ID = A.VER_ID) )
		ORDER BY A.ROUT_POINT_ID
